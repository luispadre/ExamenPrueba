CREATE DATABASE Padre;
USE Padre;

CREATE TABLE `DatosPersonales` (
	`INE` VARCHAR(50) NOT NULL,
	`Nombre` VARCHAR(50) NOT NULL,
	`APP` VARCHAR(50) NOT NULL,
	`APM` VARCHAR(50) NOT NULL,
	`Estado` VARCHAR(50) NOT NULL,
	`Telefono` VARCHAR(13) NOT NULL DEFAULT 0,
	PRIMARY KEY (`INE`)
);

CREATE TABLE `OTRO` (
	`INE` VARCHAR(50) NOT NULL DEFAULT '',
	`Colores` VARCHAR(50) NULL DEFAULT NULL,
	`Hobby` VARCHAR(50) NULL DEFAULT NULL,
	CONSTRAINT `FK__datospersonales` FOREIGN KEY (`INE`) REFERENCES `datospersonales` (`INE`)
);
